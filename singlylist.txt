#include<iostream>
using namespace std;

class node
{
    public:
    int data;
    node *next;
    node (int x)
    {
        data = x;
        next = NULL;
    }
};


class link_list
{
    public :
    node *head;
    link_list()
    {
        head = NULL ;

    }
    void insert (int x)
    {
        node *p = new node(x);
        if (head == NULL)
        {
            head = p;
        }
        else{
            node *q = head;
            while (q -> next != NULL)
            {
                q = q -> next ; 

            }
            q -> next = p;
        }
    }
      

    void remove (int x)
    {
        node *p = NULL;
        node *q = head;
        if (head -> data == x)
        {
            p = head ;
            head = head -> next ;
            delete p;
            return ;
        }
        
        
            while (q != NULL)
            {
                if (q -> next -> data == x)
                {
                    p = q -> next ;
                    q -> next = p -> next ;
                    delete p;
                }
                q = q -> next;
            }

        

    }


    void display (void)
    {
        node *q = head ;
        while (q != NULL)
        {
            cout << q -> data << "->" ;
            q = q -> next ;
        }
    }



};


int main ()
{
    link_list list;
    list.insert(10);
    list.insert(20);
    list.insert(30);
    list.insert(40);
    list.insert(50);
    list.remove(40);
    list.display();
    
    return 0;
}
