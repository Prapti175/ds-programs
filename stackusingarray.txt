#include<iostream>
using namespace std;

const int MAX = 5 ;

class Stack
{
private:
    int a[MAX];
    int top;

public:
    Stack()
    {
        top = -1;
    }

    void push(int x)
    {
        a[++top] = x;  // Fix: change '=' to '++top' before assignment
    }

    int pop(void)
    {
        int x = a[top];
        top--;
        return (x);
    }

    int peek(void)
    {
        return (a[top]);
    }

    int full(void)
    {
        if (top == MAX - 1)
        {
            return (1);
        }
        else
        {
            return (0);   // Fix: return 0 when not full instead of 1
        }
    }

    int empty(void)
    {
        if (top == -1)
        {
            return (1);
        }
        else
        {
            return (0);
        }
    }

    void display(void)
    {
        for (int i = top; i >= 0; i--)
            cout << endl << a[i];
    }
};

int main()
{
    int x;
    Stack s;
    if (!s.full())
    {
        s.push(10);
    }
    if (!s.full())
    {
        s.push(20);
    }
    if (!s.full())
    {
        s.push(30);
    }
    if (!s.full())
    {
        s.push(40);
    }
    if (!s.full())
    {
        s.push(50);
    }
    if (!s.full())
    {
        s.push(60);
    }
    if (!s.empty())
    {
        x = s.pop();
        cout << endl << "Remove : " << x;
        x = s.pop();
        cout << endl << "Remove : " << x;
    }

    s.display();
    x = s.peek();
    cout << endl << "Top element is : " << x;

    return 0;
}
